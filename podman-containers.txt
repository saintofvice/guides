# CONFIGURATION
## INITIALIZATION (ONLY NON-LINUX -BSDS, MACOS, WINDOWS-)
# create VM on OS that dont support Linux containers natively (BSDs, MacOS, Windows)
podman machine init
# pulls a Fedora image
# allocates disk space
# registers the machine in "~/.local/share/containers/podman/machine/"

# resources can be set:
podman machine init --cpus 2 --memory 2048 --disk-size 20
# 2 CPUs
# 2 GB RAM
# 20 GB disk

# start the VM (NOT yet the container)
podman machine start

# verify
podman info

## repos
# add unqualified registries (wont pull otherwise)
echo 'unqualified-search-registries = ["docker.io"]' >>  /etc/containers/registries.conf.d/10-unqualified-search-registries.conf

## PERMISSIONS
# only root can run pods by default

# check permissions
$ sysctl kernel.unprivileged_userns_clone
# Rootless is enabled if "1"

# Enable rootless for user <USER>
usermod --add-subuids 100000-165535 --add-subgids 100000-165535 <USER>

# as user
$ podman system migrate

# COMMANDS
## GENERAL
# search container
podman search <CONTAINER>

# pull an image
podman pull <CONTAINER>

# remove
podman rm

# list downloaded
podman images

# list running
podman ps

# logs
podman logs -l

# PIDs
podman top -l

# check overlay
podman info | grep -i overlay

## RUNNING
podman run --rm --os=<OS> <CONTAINER> <COMMAND>
# "--rm" removes after run
# "--os=<OS>" defines an OS to run under
# "<CONTAINER>" is the name of container to be pulled
# "<COMMAND>" the command to run inside the container
# i.e.
podman run --rm --os=linux alpine cat /etc/os-release | head 1

# "-it": interactive and set a tty (to get a shell)
podman run -it --rm --os=<OS> <CONTAINER> <PATH_TO_SHELL>
# i.e.
podman run -it --rm --os=linux alpine /bin/sh

## SYSTEMD INTEGRATION
# generate a systemd unit (all as user, not root)
# previously "mkdir -p ~/.config/systemd/user/"
podman generate system --new --name <CONTAINER> > ~/.config/systemd/user/<CONTAINER>.service
chmod 644 ~/.config/systemd/user/<CONTAINER>.service
# restart systemd
systemctl --user daemon-reload
# enable on boot
systemctl --user enable <CONTAINER>.service
# check status
systemctl --user status  <CONTAINER>.service
