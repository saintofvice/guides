# Create.
# Usage: ginit
ginit(){
	git config --global user.email "USER@MAIL.COM" ;
	git config --global user.name "USER" ;
	git init ;
	git add . ;
	git commit -m "Initial commit" ;
	git remote add origin git@github.com:USER/"$(basename $(pwd))" ;
}
# Commit.
# Usage: gommit "COMMENT"
gommit(){
	git add . ;
	git commit -m "$1" ;
}
# Push to Github.
# Usage: ghub
ghub(){
	# If you changed just a branch, you need to push that branch (i.e. "main").
	if test -z "$1" ; then
		GIT_BRANCH="master"
	else
		GIT_BRANCH="$1"
	fi
	git remote set-url origin git@github.com:USER/"$(basename $(pwd))" ;
	eval "$(ssh-agent -s)" ;
	ssh-add ~/.ssh/id_ed25519 ;
	git push -u origin "$GIT_BRANCH" ;
}
